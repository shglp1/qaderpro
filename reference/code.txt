// ─── LOAD / SAVE ─────────────────────────
function getData(key, def) {
  try { const v = localStorage.getItem(key); return v ? JSON.parse(v) : def; }
  catch { return def; }
}
function setData(key, val) { localStorage.setItem(key, JSON.stringify(val)); }

  });

// Stat counters
document.querySelectorAll('.stat-number').forEach(el => {
  const target = +el.dataset.target;
  ScrollTrigger.create({
    trigger: el,
    start: 'top 85%',
    once: true,
    onEnter: () => {
      gsap.to({ val: 0 }, {
        val: target, duration: 1.5, ease: 'power2.out',
        onUpdate: function() { el.textContent = Math.round(this.targets()[0].val) + '+'; }
      });
    }
  });
});

let currentFilter = 'all';

function renderPortfolio() {
  const grid = document.getElementById('portfolio-grid');

  grid.innerHTML = filtered.map(p => `

      <div class="w-full h-full flex items-end justify-start p-0" style="background:${p.color || '#1a1a1a'}">

// Filter buttons
document.getElementById('filter-btns').addEventListener('click', (e) => {
  const btn = e.target.closest('.filter-btn');
  if (!btn) return;
  document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
  btn.classList.add('active');

  renderPortfolio();
});

// ─── MODAL ───────────────────────────────
function openModal(id) {
  const p = portfolioItems.find(x => x.id === id);
  if (!p) return;
  document.getElementById('modal-title').textContent = p.title;
  document.getElementById('modal-desc').textContent = p.desc;
  document.getElementById('modal-video-url').textContent = p.video || '—';
  document.getElementById('modal-overlay').classList.add('open');
  document.body.style.overflow = 'hidden';
}

function closeModal(e) {
  if (!e || e.target === document.getElementById('modal-overlay') || !e.currentTarget) {
    document.getElementById('modal-overlay').classList.remove('open');
    document.body.style.overflow = '';
  }
}

// ─── ADMIN ───────────────────────────────
function openAdmin() {
  loadAdminFields();
  loadAdminPortfolio();
  document.getElementById('admin-panel').classList.add('open');
  document.body.style.overflow = 'hidden';

function closeAdmin() {
  document.getElementById('admin-panel').classList.remove('open');

}

function switchTab(name, btn) {
  document.querySelectorAll('.admin-tab-content').forEach(t => t.classList.add('hidden'));
  document.querySelectorAll('.admin-tab-btn').forEach(b => {
    b.classList.remove('text-gold', 'border-gold');
    b.classList.add('text-gray-500', 'border-transparent');
  });
  document.getElementById('tab-' + name).classList.remove('hidden');
  btn.classList.add('text-gold', 'border-gold');
  btn.classList.remove('text-gray-500', 'border-transparent');
}

function loadAdminFields() {
  const c = getData('qader_content', {});
  document.getElementById('a-hero-title').value = c.heroTitle || 'إبداع يليق بكل لحظة';
  document.getElementById('a-hero-sub').value = c.heroSub || 'نحوّل أفكارك إلى تجارب بصرية إبداعية تجعل علامتك التجارية تُرى وتُتذكر';
  document.getElementById('a-about-text').value = c.aboutText || document.getElementById('about-text').textContent;
  const ci = getData('qader_contact', {});
  document.getElementById('a-whatsapp').value = ci.whatsapp || '';
  document.getElementById('a-email').value = ci.email || '';
  document.getElementById('a-instagram').value = ci.instagram || '';
  document.getElementById('a-twitter').value = ci.twitter || '';
}

function saveContent() {
  setData('qader_content', {
    heroTitle: document.getElementById('a-hero-title').value,
    heroSub: document.getElementById('a-hero-sub').value,
    aboutText: document.getElementById('a-about-text').value

  // Apply live

  showSaveNotice();
}

}

// Admin portfolio list
function loadAdminPortfolio() {
  const list = document.getElementById('portfolio-admin-list');
  list.innerHTML = portfolioItems.map((p, i) => `
    <div class="glass-card p-5 rounded-2xl flex items-center gap-4">
      <div class="w-12 h-12 rounded-xl flex items-center justify-center flex-shrink-0 text-gold" style="background:${p.color}66">
        <i class="fa fa-film"></i>
      </div>
      <div class="flex-1 grid grid-cols-2 gap-3">
        <div>
          <label class="admin-label">العنوان</label>
          <input type="text" value="${p.title}" class="admin-input" onchange="updatePortfolioItem(${p.id},'title',this.value)" />
        </div>
        <div>
          <label class="admin-label">التصنيف</label>
          <select class="admin-input" onchange="updatePortfolioItem(${p.id},'cat',this.value)">
            <option value="commercial" ${p.cat==='commercial'?'selected':''}>إعلانات تجارية</option>
            <option value="corporate" ${p.cat==='corporate'?'selected':''}>أفلام تعريفية</option>
            <option value="events" ${p.cat==='events'?'selected':''}>فعاليات</option>
            <option value="digital" ${p.cat==='digital'?'selected':''}>محتوى رقمي</option>
          </select>
        </div>
        <div class="col-span-2">
          <label class="admin-label">رابط الفيديو</label>
          <input type="text" value="${p.video || ''}" placeholder="https://..." class="admin-input" onchange="updatePortfolioItem(${p.id},'video',this.value)" />
        </div>
      </div>
      <button onclick="deletePortfolioItem(${p.id})" class="text-red-400 hover:text-red-300 text-sm transition-colors flex-shrink-0">
        <i class="fa fa-trash"></i>
      </button>
    </div>
  `).join('');
}

function updatePortfolioItem(id, key, val) {
  portfolioItems = portfolioItems.map(p => p.id === id ? {...p, [key]: val} : p);
  setData('qader_portfolio', portfolioItems);
  renderPortfolio();
}

function deletePortfolioItem(id) {
  portfolioItems = portfolioItems.filter(p => p.id !== id);
  setData('qader_portfolio', portfolioItems);
  loadAdminPortfolio();
  renderPortfolio();
}

function addPortfolioItem() {
  const newItem = {
    id: Date.now(), title: 'مشروع جديد', cat: 'commercial',
    tag: 'إعلان تجاري', desc: 'وصف المشروع', color: '#1a1a2a', video: ''
  };
  portfolioItems.push(newItem);
  setData('qader_portfolio', portfolioItems);
  loadAdminPortfolio();
  renderPortfolio();
}
